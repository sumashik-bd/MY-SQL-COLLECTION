FOR GIVE RESTRICTION INSERT/UPDATE IN ITEMS:
SET_ITEM_PROPERTY(‘EMPLOYEES.EMP_ID’,INSERT_ALLOWED,PROPERTY_FALSE);
SET_ITEM_PROPERTY(‘EMPLOYEES.EMP_ID’,UPDATE_ALLOWED,PROPERTY_FALSE);
FOR REQUIRED PROPERTY:
SET_ITEM_PROPERTY(‘EMPLOYEES.EMP_ID’,REQUIRED,PROPERTY_FALSE);
FOR ENABLED/DISABLED ITEM:
SET_ITEM_PROPERTY(‘EMPLOYEES.EMP_ID’,ENABLED,PROPERTY_FALSE);
FOR ANY COLOR TO ITEM:
SET_ITEM_INSTANCE_PROPERTY(‘EMPLOYEES.EMP_ID’, CURRENT_RECORD,VISUAL_ATTRIBUTE,’VISUAL_ATTRIBUTE_NAME’);
FOR TOOLTIP TEXT FOR A TEXT ITEM.
SET_ITEM_PROPERTY(‘EMPLOYEES.FIRST_NAME’, TOOLTIP_TEXT, ‘WRITE HERE FIRST NAME’);
FOR FONT BOLD AN ITEMS.
SET_ITEM_PROPERTY(‘EMPLOYEES.SUSPENSE’, FONT_WEIGHT, FONT_BOLD);
 
FOR CHANGE THE WIDTH OF AN ITEM (PUSH BUTTON,TEXT ITEMS).
SET_ITEM_PROPERTY(‘EMPLOYEES.FIRST_NAME’, WITDTH,189);
FOR CHANGE THE PROMPT TEXT OF AN ITEM IN ORACLE FORMS.
SET_ITEM_PROPERTY(‘EMPLOYEES.FIRST_NAME’, PROMPT_TEXT, ‘FIRST_NAME:’);
 
FOR TO CHANGE THE LABEL OF A PUSH BUTTON.
SET_ITEM_PROPERTY(‘CNTRL.SAVE’, LABEL, SAVE_EMPLOYEE’);
