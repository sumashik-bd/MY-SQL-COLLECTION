If :BTN_BLOCK.SEARCH IS NOT NULL THEN
		DECLARE
			V_REC NUMBER;
		BEGIN
			GO_BLOCK('EOMS_REQ_PREP_MST');
			LAST_RECORD;
		   	V_REC := :SYSTEM.CURSOR_RECORD; 
		   	FIRST_RECORD;
		   	FOR I in 1..V_REC LOOP
			IF  UPPER(:EOMS_REQ_PREP_MST.ITEM) LIKE ''||UPPER(:BTN_BLOCK.SEARCH)||'%' THEN
			GO_ITEM('EOMS_REQ_PREP_MST.ITEM');
			EXIT;
			END IF;
		   	NEXT_RECORD;
		   	END LOOP;
   	END;
END IF;
